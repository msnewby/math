  function setFilter(f, btn) {

    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    applyFilters();
  }

  function filterCards() { applyFilters(); }

  function applyFilters() {
    const q = document.getElementById('search').value.toLowerCase();
    document.querySelectorAll('.lesson-card, .coming-card').forEach(card => {
      const gradeMatch = activeFilter === 'all' || card.dataset.grade === activeFilter || activeFilter === card.dataset.status;
      const readyMatch = activeFilter === 'ready' ? card.dataset.status === 'ready' : true;
      const textMatch = !q || card.innerText.toLowerCase().includes(q);
      card.style.display = (gradeMatch && readyMatch && textMatch) ? '' : 'none';
    });
    document.querySelectorAll('.section-header').forEach(h => {
      const grade = h.dataset.grade;
      const visible = [...document.querySelectorAll(`.lesson-card[data-grade="${grade}"], .coming-card[data-grade="${grade}"]`)]
        .some(c => c.style.display !== 'none');
      h.style.display = visible ? '' : 'none';
    });
